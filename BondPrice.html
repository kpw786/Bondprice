class BondDataLoader:
    def __init__(self, file_mappings: dict, roc_date: str, priority: dict = None):
        """
        file_mappings: dict, 
            {
                ".xlsx": {"BondID": "A", "Yield": "B", "Price": "C", "Source": "", "type": "xlsx"},
                ".xls": {"BondID": "D", "Yield": "E", "Price": "F", "Source": "", "type": "xls"},
            }
        roc_date: str,  ()
        priority: dict, () 
        """
        self.file_mappings = file_mappings
        self.roc_date = roc_date
        self.priority = priority if priority else {}
        self.all_data = pd.DataFrame()

    def _col_to_index(self, col: str) -> int:
        """ Excel  index (A=0, B=1, ...)"""
        return ord(col.upper()) - ord("A")

    def _read_file(self, file: str, file_type: str):
        """"""
        if file_type == "xlsx":
            return pd.read_excel(file, engine="openpyxl", header=None)
        elif file_type == "xls":
            return pd.read_excel(file, engine="xlrd", header=None)
        elif file_type == "csv":
            return pd.read_csv(file, encoding="cp950", header=None)
        else:
            raise ValueError(f"Unknown file type for {file}")

    def _process_file(self, file: str, cols: dict):
        """"""
        # 
        col_map = {name: self._col_to_index(col) 
                   for name, col in cols.items() if name not in ["Source", "type"]}

        # 
        df = self._read_file(file, cols["type"])

        # 
        if cols["Source"] == "":
            date_col = self._col_to_index("J")  # J 
            df = df[df.iloc[:, date_col].astype(str).str.strip() == self.roc_date]
            print(f" {len(df)}  {self.roc_date} ")

        # 
        temp = df.iloc[:, [col_map["BondID"], col_map["Yield"], col_map["Price"]]].copy()
        temp.columns = ["BondID", "Yield", "MarketPrice"]
        temp["Source"] = cols["Source"]

        return temp

    def load_all(self):
        """"""
        for file, cols in self.file_mappings.items():
            temp = self._process_file(file, cols)
            self.all_data = pd.concat([self.all_data, temp], ignore_index=True)
        return self.all_data

class BondDVO1Calculator:
    def __init__(self, df: pd.DataFrame):
        self.df = df  
        self.dvo1_by_type = None

    def clean_numeric_columns(self, columns: list):
        """ NaN"""
        for col in columns:
            self.df[col] = pd.to_numeric(self.df[col], errors="coerce")

    def calculate_dvo1(self):
        """ DVO1  DataFrame"""
        self.df["DVO1"] = (
            self.df["Duration"] * 
            self.df["MarketPrice"] * 
            0.0001 * 
            self.df["Position(Mio)"]
        )
        #  NaN  0
        self.df["DVO1"] = self.df["DVO1"].fillna(0)

    def aggregate_by_type(self):
        """ BondType  DVO1"""
        self.dvo1_by_type = self.df.groupby("BondType")["DVO1"].sum()
        return self.dvo1_by_type

    def run(self):
        self.clean_numeric_columns(["Duration", "MarketPrice", "Position(Mio)"])
        self.calculate_dvo1()
        return self.aggregate_by_type()

class KruskalWallisTest:
    def __init__(self, group_data: dict):
        """
        group_data: dict, keyvalue
         {"": corp_dur, "": gov_dur, "": fi_dur}
        """
        self.group_data = group_data
        self.h_stat = None
        self.p_value = None

    def run_test(self):
        #  kruskal
        self.h_stat, self.p_value = stats.kruskal(*self.group_data.values())

    def report(self):
        group_names = ", ".join(self.group_data.keys())
        print(f"Kruskal-Wallis H-test for {group_names}: H={self.h_stat}, p-value={self.p_value}")

class MannWhitneyTest:
    def __init__(self, group1_name, group1_data, group2_name, group2_data):
        self.group1_name = group1_name
        self.group1_data = group1_data
        self.group2_name = group2_name
        self.group2_data = group2_data
        self.u_stat = None
        self.p_val = None

    def run_test(self, alternative='greater'):
        self.u_stat, self.p_val = stats.mannwhitneyu(
            self.group1_data, self.group2_data, alternative=alternative
        )

    def report(self):
        print(f"{self.group1_name} vs {self.group2_name}: U={self.u_stat}, p={self.p_val}")
