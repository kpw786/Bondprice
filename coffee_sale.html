# Time of Day 
def time_of_day(hour):
    if pd.isna(hour):
        return None
    elif 5 <= hour < 12:
        return "Morning"
    elif 12 <= hour < 17:
        return "Afternoon"
    else:
        return "Night"

df["Time_of_Day"] = df["hour_of_day"].apply(time_of_day)

def coffee_category(name):
    espresso_based = ["Latte", "Americano", "Cappuccino", "Espresso","Americano with Milk","Cortado"]
    if name in espresso_based:
        return "Espresso-based"
    elif "Hot Chocolate" in name or "Cocoa" in name:
        return "Non-coffee"
    else:
        return "Other"

df["Coffee_category"] = df["coffee_name"].apply(coffee_category)
